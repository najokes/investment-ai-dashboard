Пример кода

import quantumai as qai

# Инициализация клиента 2026
client = qai.Client(
    api_key="qai_2026_quantum_key",
    quantum_mode=True
)

# Квантовый прогноз LTV
prediction = client.predict_ltv(
    customer_data={...},
    use_quantum=True,
    timestamp="2026-01-31T14:30:00Z"
)

# Вывод результатов
print(f"Predicted LTV: {prediction.value:.2f}M ₽")
print(f"Quantum Confidence: {prediction.confidence:.1%}")
